    <div id="quiz-container" class="max-w-2xl mx-auto bg-white p-8 rounded-lg shadow-md">
        <h1 class="text-2xl font-bold mb-4 text-blue-600">📖 學生自主練習系統</h1>
        <div id="question-area">
            <p id="q-text" class="text-lg font-semibold mb-4 text-gray-800">載入中...</p>
            <div id="options" class="space-y-3"></div>
        </div>
        <div id="result-area" class="hidden text-center">
            <h2 class="text-xl font-bold mb-4">測驗結束！</h2>
            <p id="score" class="text-3xl text-blue-500 mb-4"></p>
            <button onclick="location.reload()" class="bg-blue-500 text-white px-6 py-2 rounded">重新挑戰</button>
        </div>
    </div>

    <script>
        // 這裡可以填入你的題目
        const quizData = [
            { q: "HTML 代表什麼？", a: ["超文字標記語言", "高科技語言", "排版語言"], correct: 0 },
            { q: "哪個是網頁的樣式表語言？", a: ["Python", "Java", "CSS"], correct: 2 }
        ];

        let currentQ = 0;
        let score = 0;

        function loadQuestion() {
            const q = quizData[currentQ];
            document.getElementById('q-text').innerText = `${currentQ + 1}. ${q.q}`;
            const optDiv = document.getElementById('options');
            optDiv.innerHTML = '';
            q.a.forEach((opt, i) => {
                const btn = document.createElement('button');
                btn.innerText = opt;
                btn.className = "w-full text-left p-3 border rounded hover:bg-blue-50 transition";
                btn.onclick = () => checkAnswer(i);
                optDiv.appendChild(btn);
            });
        }

        function checkAnswer(idx) {
            if(idx === quizData[currentQ].correct) score++;
            currentQ++;
            if(currentQ < quizData.length) {
                loadQuestion();
            } else {
                showResult();
            }
        }

        function showResult() {
            document.getElementById('question-area').classList.add('hidden');
            document.getElementById('result-area').classList.remove('hidden');
            document.getElementById('score').innerText = `得分：${score} / ${quizData.length}`;
        }

        loadQuestion();
    </script>
